function n_3_col(A::Array)
   S = size(A)
   p_col = mean(A, 1)
   R = zeros(A)
   for j in 1:S[2]
      for i in 1:S[1]
         R[i,j] = int(rand() <= p_col[j])
      end
   end
   return R
end
